81template <typename T>
82inline bool intersects(geom::segment2<T> const& lhs, geom::segment2<T> const& rhs)
83{
84 math::vec2<T> const& a = lhs.a;
85 math::vec2<T> const& b = lhs.b;
86 math::vec2<T> const& c = rhs.a;
87 math::vec2<T> const& d = rhs.b;
88
89 // compute which side of the lhs each point is on
90 T const c_side = math::orientation(a, b, c);
91 T const d_side = math::orientation(a, b, d);
92
93 if (c_side == math::constants<T>::zero && d_side == math::constants<T>::zero) // if all four points are colinear
94 {
95 bool const x_overlap = intersects(lhs.interval(0), rhs.interval(0));
96 bool const y_overlap = intersects(lhs.interval(1), rhs.interval(1));
97 return x_overlap && y_overlap;
98 }
99 else // the general case where not all four points are colinear
100 {
101 T const a_side = math::orientation(c, d, a);
102 T const b_side = math::orientation(c, d, b);
103 return (c_side * d_side <= math::constants<T>::zero) && (a_side * b_side <= math::constants<T>::zero);
104 }
105}

114template <typename T>
115bool intersects(geom::segment2<T> const& segment, geom::aabb2<T> const& aabb)
116{
117 // if either point is in the box, there is certainly an intersection
118 if (aabb.contains(segment.a) || aabb.contains(segment.b))
119 {
120 return true;
121 }
122
123 // both points are outside the box => an intersection can only occur if the segment crosses a diagonal
124 geom::segment2<T> const diagonal_0(aabb.min, aabb.max);
125 if (intersects(segment, diagonal_0))
126 {
127 return true;
128 }
129
130 math::vec2<T> const a(aabb.min.x, aabb.max.y);
131 math::vec2<T> const b(aabb.max.x, aabb.min.y);
132 geom::segment2<T> const diagonal_1(a, b);
133 if (intersects(segment, diagonal_1))
134 {
135 return true;
136 }
137
138 return false; // fallthrough to return false
139}

321template <typename T>
322bool intersects(geom::ray3<T> const& ray, geom::plane<T> const& plane)
323{
324 T const signed_dist = plane.signed_dist(ray.origin);
325 T const dot = ray.direction.dot(plane.normal());
326 if (signed_dist ==
327 math::constants<T>::zero) // if the origin of the ray is on the plane, then they certainly intersect
328 {
329 return true;
330 }
331 else if (dot == math::constants<T>::zero) // if the normal and the ray are orthogonal, they do not intersect
332 {
333 return false;
334 }
335 else
336 {
337 // we now know that the origin of the ray is not on plane and the ray is not parallel to the
338 // plane. this means that both signed_dist and dot are non-zero. if the signs of those values
339 // are opposite, then the ray and the plane intersect
340 return std::signbit(signed_dist) != std::signbit(dot);
341 }
342}
