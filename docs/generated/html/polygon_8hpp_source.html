137 {
138 // early out for malformed polygons
139 if (m_vertices.size() < 3)
140 {
141 return false;
142 }
143
144 size_t clockwise = 0;
145 size_t counterclockwise = 0;
146
147 size_t size = m_vertices.size();
148 for (size_t i = 0; i < size; ++i)
149 {
150 // grab three consecutive points
151 vec_t const& a = m_vertices[(i + 0) % size];
152 vec_t const& b = m_vertices[(i + 1) % size];
153 vec_t const& c = m_vertices[(i + 2) % size];
154 T orientation = math::orientation(a, b, c);
155
156 // update counts
157 if (orientation > math::constants<T>::zero)
158 {
159 ++counterclockwise;
160 }
161 if (orientation < math::constants<T>::zero)
162 {
163 ++clockwise;
164 }
165
166 // if we have different orientations, then the polygon is not convex

171 }

215 bool contains(vec_t const& query, boundary_types const type) const
216 {
217 // we shoot a ray out from point in +x and count the number of edges that are crossed
218 // crossing_count is odd => point is in polygon
219 // crossing_count is even => point is not in polygon
220 size_t crossing_count = 0;
221
222 // early out for malformed polygons and bbox query
223 if (m_vertices.size() < 3)
224 {
225 return false;
226 }
227 if (!m_aabb.contains(query))
228 {
229 return false;
230 }
231
232 // iterate over all edges, computing if the ray crosses the edge
233 for (size_t i = 0; i < m_vertices.size(); ++i)
234 {
235 geom::segment2<T> seg = edge(i);
236 if (geom::dist_squared(seg, query) == math::constants<T>::zero) // early out if the point is on the boundary
237 {
238 return type == boundary_types::closed;
239 }
240 else if ((seg.a.y > query.y) != (seg.b.y > query.y)) // test if the y-range is relevent
241 {
242 math::interval<T> x_range = seg.interval(0);
243 if (query.x < x_range.a)
244 {
245 ++crossing_count;
246 } // avoid floating point work if we know the segment crosses
247 else if (query.x <= x_range.b) // make sure the x-range is relevant
248 {
249 if (seg.a.x == seg.b.x) // check for a vertical line
250 {
251 if (query.x < seg.a.x)
252 {
253 ++crossing_count;
254 }
255 }
256 else
257 {
258 // we compute the x coordinate of the pair (x, query.y) that is on the line defined by seg
259 // if the ray begins before that coordinate, then the ray crosses this segment
260 T const slope_inv = math::constants<T>::one / seg.slope();
261 T const x = slope_inv * (query.y - seg.a.y) + seg.a.x;
262 if (query.x < x)
263 {
264 ++crossing_count;
265 }
266 }
267 }
268 }
269 }
270
271 // crossing_count is odd => point is in polygon
272 // crossing_count is even => point is not in polygon
273 return crossing_count % 2 == 1;
274 }
