202 // move the position to the next available interval that might contain the query (jumps nodes if necessary)
203 void advance(T const query)
204 {
205 if (node)
206 {
207 center_t const& center = node->center;
208 if (query <= center.pivot) // check if we need to jump nodes
209 {
210 // if any of the following are true, we jump nodes
211 // 1. this iterator is the end (checks the edge case where we constructed a position that
212 // points to an empty node)
213 // 2. the iterator points to a range that does not contain the query
214 // 3. the incremented iterator is the end
215 if (it == center.lesser.end() || query < (*it)->interval.a || ++it == center.lesser.end())
216 {
217 jump(query);
218 }
219 }
220 else
221 {
222 // if any of the following are true, we jump nodes
223 // 1. this iterator is the end (checks the edge case where we constructed a position that
224 // points to an empty node)
225 // 2. the iterator points to a range that does not contain the query
226 // 3. the incremented iterator is the greater
227 if (it == center.greater.end() || query > (*it)->interval.b || ++it == center.greater.end())

431 static std::unique_ptr<node_t> construct(factory_args const& args)
432 {
433 // base case where there are no intervals
434 if (args.intervals.empty())
435 {
436 return nullptr;
437 }
438
439 // args.sorted.size() will always be even because it will have exactly 2 * args.intervals.size() elements -- two
440 // endpoints per interval
441 size_t mid = args.sorted.size() / 2;
442 T median = math::constants<T>::half * (args.sorted[mid - 1].x + args.sorted[mid].x);
443
444 factory_args left{};
445 factory_args right{};
446 center_t center(median);
447
448 // iterate over intervals, adding to the appropriate node
449 for (entry_ptr_t entry : args.intervals)
450 {
451 if (entry->interval.b < center.pivot)
452 {
453 left.intervals.push_back(entry);
454 } // interval is entirely to the left
455 else if (center.pivot < entry->interval.a)
456 {
457 right.intervals.push_back(entry);
458 } // interval is entirely to the right
459 else
460 {
461 center.insert(entry);
462 } // invterval contains the pivot
463 }
464
465 // iterate over sorted points, adding to the appropriate node
466 for (sortable_t sortable : args.sorted)
467 {
468 if (sortable.entry->interval.b < center.pivot)
469 {
470 left.sorted.push_back(sortable);
471 } // interval is entirely to the left
472 else if (center.pivot < sortable.entry->interval.a)
473 {
474 right.sorted.push_back(sortable);
475 } // interval is entirely to the right
476 }
477
478 center.sort(); // sort the intervals that contain the pivot
479
480 return std::make_unique<node_t>(std::move(center), interval_tree::construct(left),
481 interval_tree::construct(right));
482 }
