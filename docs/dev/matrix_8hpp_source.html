1105template <typename T>
1106inline mtx4<T> rotate(vec<T, 3> const& axis, T const theta)
1107{
1108 // perform computations once
1109 T const cosine = std::cos(theta);
1110 T const sine = std::sin(theta);
1111 T const comp = T(1) - cosine; // complement of the cosine
1112
1113 // local variables for less verbose code
1114 T const x = axis.x;
1115 T const y = axis.y;
1116 T const z = axis.z;
1117
1118 // compute row vectors
1119 vec<T, 4> row_0(x * x * comp + cosine, x * y * comp - z * sine, x * z * comp + y * sine, T(0));
1120 vec<T, 4> row_1(y * x * comp + z * sine, y * y * comp + cosine, y * z * comp - x * sine, T(0));
1121 vec<T, 4> row_2(x * z * comp - y * sine, y * z * comp + x * sine, z * z * comp + cosine, T(0));
1122
1123 // construct return matrix
1124 mtx4<T> rotation;
1125 rotation.row(0) = row_0;
1126 rotation.row(1) = row_1;
1127 rotation.row(2) = row_2;
1128 return rotation;
