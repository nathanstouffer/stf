424 T determinant() const

1104template <typename T>
1105inline mtx4<T> rotate(vec<T, 3> const& axis, T const theta)
1106{
1107 // perform computations once
1108 T const cosine = std::cos(theta);
1109 T const sine = std::sin(theta);
1110 T const comp = T(1) - cosine; // complement of the cosine
1111
1112 // local variables for less verbose code
1113 T const x = axis.x;
1114 T const y = axis.y;
1115 T const z = axis.z;
1116
1117 // compute row vectors
1118 vec<T, 4> row_0(x * x * comp + cosine, x * y * comp - z * sine, x * z * comp + y * sine, T(0));
1119 vec<T, 4> row_1(y * x * comp + z * sine, y * y * comp + cosine, y * z * comp - x * sine, T(0));
1120 vec<T, 4> row_2(x * z * comp - y * sine, y * z * comp + x * sine, z * z * comp + cosine, T(0));
1121
1122 // construct return matrix
1123 mtx4<T> rotation;
1124 rotation.row(0) = row_0;
1125 rotation.row(1) = row_1;
1126 rotation.row(2) = row_2;
1127 return rotation;
1128}
