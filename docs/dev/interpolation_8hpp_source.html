182template <typename T>
183inline T cubic_hermite_spline(T const p0, T const m0, T const p1, T const m1, T const t)
184{
185 T t_squared = t * t;
186 T t_cubed = t_squared * t;
187
188 // compute basis values
189 T b0 = T(2) * t_cubed - T(3) * t_squared + T(1);
190 T b1 = t_cubed - T(2) * t_squared + t;
191 T b2 = T(-2) * t_cubed + T(3) * t_squared;
192 T b3 = t_cubed - t_squared;
193
194 return b0 * p0 + b1 * m0 + b2 * p1 + b3 * m1;
195}
