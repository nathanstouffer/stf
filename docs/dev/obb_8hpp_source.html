319 // intersection is computed using the separating axis test. if two 2d obbs do not intersect, then the set of edge
320 // normals will contain a separating axis

345template <typename T>
346bool intersects(obb3<T> const& lhs, obb3<T> const& rhs)
347{
348 // intersection is computed using the separating axis test. if two 3d obbs do not intersect, then the union of the
349 // following two sets will contain a separating axis
350 // 1. the set of face normals
351 // 2. the set of cross products of pairs of edges from each obb
352 for (math::vec3<T> const& axis : lhs.basis())
353 {
354 if (separates(axis, lhs, rhs))
355 {
356 return false;
357 }
358 }
359 for (math::vec3<T> const& axis : rhs.basis())
360 {
361 if (separates(axis, lhs, rhs))
362 {
363 return false;
364 }
365 }
366 for (math::vec3<T> const& l : lhs.basis())
367 {
368 for (math::vec3<T> const& r : rhs.basis())
369 {
370 math::vec3<T> const axis = math::cross(l, r);
371 if (axis.length_squared() > math::constants<T>::zero && separates(axis, lhs, rhs))
372 {
373 return false;
374 }
375 }
376 }
377 return true; // fallthrough to true
378}
