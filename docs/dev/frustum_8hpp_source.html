157 // check if any box plane entirely excludes the frustum
158 {
159 // clang-format off
160 std::array<vec_t, 8> verts = {
161 // near points // far points
162 m_vertices.ntl, m_vertices.ntr, m_vertices.ftl, m_vertices.ftr,
163 m_vertices.nbl, m_vertices.nbr, m_vertices.fbl, m_vertices.fbr,
164 };
165 // clang-format on

167 for (size_t d = 0; d < 3; ++d) // iterate over each dimension (x, y, and z)
168 {
169 // check the plane at the minimum of the box that has a normal in the +d direction

304 explicit vertices(scamera<T> const& camera)
305 {
306 // compute camera basis vectors
307 vec_t look = camera.look();
308 vec_t up = camera.up();
309 vec_t right = camera.right();
310
311 // compute points on the near plane
312 {
313 T half_height = camera.nearp * std::tan(math::constants<T>::half * camera.fov);
314 T half_width = half_height * camera.aspect;
315
316 vec_t center = camera.eye + look * camera.nearp;
317 vec_t delta_up = up * half_height;
318 vec_t delta_right = right * half_width;
319
320 ntl = center + delta_up - delta_right;
321 ntr = center + delta_up + delta_right;
322 nbl = center - delta_up - delta_right;
323 nbr = center - delta_up + delta_right;
324 }
325
326 // compute points on the far plane
327 {
328 T half_height = camera.farp * std::tan(math::constants<T>::half * camera.fov);
329 T half_width = half_height * camera.aspect;
330
331 vec_t center = camera.eye + look * camera.farp;
332 vec_t delta_up = up * half_height;
333 vec_t delta_right = right * half_width;
334
335 ftl = center + delta_up - delta_right;
336 ftr = center + delta_up + delta_right;
337 fbl = center - delta_up - delta_right;
338 fbr = center - delta_up + delta_right;

355 static std::array<vec_t, 18> edge_axes(basis_t const& basis, vertices const& verts)
356 {
357 std::array<vec_t, 18> axes = {};
358 // add axes for cross products between pairs of edges from each polyhedra (the box normals and edge directions
359 // are the same)
360 for (size_t d = 0; d < 3; ++d)
361 {
362 size_t offset = d * 6; // offset by how many axes have been written to the array
363 axes[offset + 0] = math::cross(basis[d], math::normalized(verts.ftr - verts.ftl));
364 axes[offset + 1] = math::cross(basis[d], math::normalized(verts.ftr - verts.fbr));
365 axes[offset + 2] = math::cross(basis[d], math::normalized(verts.ftl - verts.ntl));
366 axes[offset + 3] = math::cross(basis[d], math::normalized(verts.ftr - verts.ntr));
367 axes[offset + 4] = math::cross(basis[d], math::normalized(verts.fbl - verts.nbl));
368 axes[offset + 5] = math::cross(basis[d], math::normalized(verts.fbr - verts.nbr));
369 }
