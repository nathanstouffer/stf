24template <typename T>
25region_code code(geom::aabb2<T> const& box, math::vec2<T> const& point)
26{
27 region_code x = region_code::inside;
28 if (point.x < box.min.x)
29 {
30 x = region_code::left;

32 else if (box.max.x < point.x)

66template <typename T>
67bool clip(geom::aabb2<T> const& box, geom::segment2<T>& seg)
68{
69 guts::region_code code0 = guts::code(box, seg.a);
70 guts::region_code code1 = guts::code(box, seg.b);
71
72 // the official algorithm uses an infinite loop here, but there are certain inputs with floating-point precision
73 // issues that cause the loop to never terminate in practice. 4 is the maximum number of iterations that should ever
74 // occur
75 for (size_t i = 0; i < 4; ++i)
76 {
77 bool outside = code0 | code1;
78 if (!outside) // if both points are inside the box, accept the segment
79 {
80 return true;
81 }
82 else if (code0 & code1) // if both points share a region, reject the segment
83 {
84 return false;
85 }
86 else // otherwise, pick a point and clip it to a line defined by the box
87 {
88 math::vec2<T> point;
89 guts::region_code code = code1 > code0 ? code1 : code0;
90 if (code & guts::region_code::top)
91 {
92 point.x = seg.a.x + (box.max.y - seg.a.y) * seg.slope_inv();
93 point.y = box.max.y;
94 }
95 else if (code & guts::region_code::bottom)
96 {
97 point.x = seg.a.x + (box.min.y - seg.a.y) * seg.slope_inv();
98 point.y = box.min.y;
99 }
100 else if (code & guts::region_code::right)
101 {
102 point.x = box.max.x;
103 point.y = seg.a.y + (box.max.x - seg.a.x) * seg.slope();
104 }
105 else if (code & guts::region_code::left)
106 {
107 point.x = box.min.x;
108 point.y = seg.a.y + (box.min.x - seg.a.x) * seg.slope();
109 }
110
111 // update the segment endpoint
112 if (code == code0)
113 {
114 seg.a = point;
115 code0 = guts::code(box, seg.a);
116 }
117 else
118 {
119 seg.b = point;
120 code1 = guts::code(box, seg.b);
121 }
122 }
123 }
124 return false;
125}

134template <typename T>
135std::vector<geom::polyline2<T>> clip(geom::aabb2<T> const& box, geom::polyline2<T> const& polyline)
136{
137 std::vector<geom::polyline2<T>> clipped;
138
139 geom::polyline2<T> sub_polyline;
140 for (size_t i = 0; i + 1 < polyline.size(); ++i)
141 {
142 geom::segment2<T> edge = polyline.edge(i);
143 if (clip(box, edge)) // segment intersects the box => test how to add the segment to the clipped result
144 {
145 if (sub_polyline.is_empty()) // this is the a fresh polyline => add both points
146 {
147 sub_polyline.push_back(edge.a);
148 sub_polyline.push_back(edge.b);
149 }
150 else if (sub_polyline.back() == edge.a) // add to existing polyline => only add the second point
151 {
152 sub_polyline.push_back(edge.b);
153 }
154 else // part of a new polyline => store the old polyline and start a new one
155 {
156 clipped.push_back(sub_polyline);
157
158 sub_polyline.clear();
159 sub_polyline.push_back(edge.a);
160 sub_polyline.push_back(edge.b);
161 }
162 }
163 else if (!sub_polyline.is_empty()) // segment does not intersect the box => immediately add the current
164 // sub_polyline to the result
165 {
166 clipped.push_back(sub_polyline);
167 sub_polyline.clear();
168 }
169 }
170
171 // edge case to check if the final sub_polyline was not added to the result
172 if (!sub_polyline.is_empty())
173 {
174 clipped.push_back(sub_polyline);
175 }
176
177 return clipped;
178}
